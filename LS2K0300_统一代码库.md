## 3.2 gpio.h — GPIO 统一接口

```cpp
/*******************************************************************************
* hal/gpio.h — Linux sysfs GPIO 封装
* 功能: 导出/释放/direction/value 读写
* 两种模式:
*   - gpio_quick: fopen/fclose 模式 (~200µs, 适合低频操作)
*   - gpio_fd:    持久文件描述符模式 (~80µs, 适合 PWM/高频读写)
*******************************************************************************/
#pragma once
#include <stdio.h>
#include <stdlib.h>
#include <fcntl.h>
#include <unistd.h>
#include <string.h>

// ====== 低频模式 (fopen/fclose) ======
struct gpio_quick {
    static void setup(int g) {
        char p[64];
        // 1. 先释放残留状态
        snprintf(p,64,"/sys/class/gpio/gpio%d/value",g);
        if(access(p,0)==0){
            FILE* f=fopen("/sys/class/gpio/unexport","w");
            if(f){fprintf(f,"%d",g);fclose(f);usleep(50000);}
        }
        // 2. 导出
        FILE* f=fopen("/sys/class/gpio/export","w");
        if(f){fprintf(f,"%d",g);fclose(f);usleep(100000);}
        // 3. 设方向为out
        snprintf(p,64,"/sys/class/gpio/gpio%d/direction",g);
        f=fopen(p,"w"); if(f){fprintf(f,"out");fclose(f);}
    }
    static void set(int g, int v) {
        char p[64]; snprintf(p,64,"/sys/class/gpio/gpio%d/value",g);
        FILE* f=fopen(p,"w"); if(f){fprintf(f,"%d",v);fclose(f);}
    }
};

// ====== 高频模式 (持久fd, 用于PWM) ======
struct gpio_fd {
    int fd;

    gpio_fd() : fd(-1) {}

    void open_rw(int g) {
        char p[64]; snprintf(p,64,"/sys/class/gpio/gpio%d/value",g);
        fd = ::open(p, O_RDWR);
    }
    void write(int v) {
        if(fd<0) return;
        char c = v ? '1' : '0';
        lseek(fd, 0, SEEK_SET);
        ::write(fd, &c, 1);
    }
    int read() {
        if(fd<0) return -1;
        char c;
        lseek(fd, 0, SEEK_SET);
        ::read(fd, &c, 1);
        return c == '1' ? 1 : 0;
    }
    void close() { if(fd>=0){ ::close(fd); fd=-1; } }
};

// ====== 微秒级忙等延时 ======
inline long long us_now() {
    struct timespec ts;
    clock_gettime(CLOCK_MONOTONIC, &ts);
    return (long long)ts.tv_sec * 1000000 + ts.tv_nsec / 1000;
}
inline void delay_us(int us) {
    long long end = us_now() + us;
    while(us_now() < end);
}
```

## 3.3 motor.h — 差速转向控制

```cpp
/*******************************************************************************
* hal/motor.h — 差速转向控制
* 控制 L298N 驱动板实现前进/后退/左转/右转/停车
*******************************************************************************/
#pragma once
#include "gpio.h"
#include "config.h"

struct motor {
    gpio_fd in1, in2, in3, in4;

    void init() {
        int pins[] = {MOTOR_L_IN1, MOTOR_L_IN2, MOTOR_R_IN1, MOTOR_R_IN2};
        for(int g : pins) gpio_quick::setup(g);
        in1.open_rw(MOTOR_L_IN1); in2.open_rw(MOTOR_L_IN2);
        in3.open_rw(MOTOR_R_IN1); in4.open_rw(MOTOR_R_IN2);
    }
    void raw(int L1, int L2, int R1, int R2) {
        in1.write(L1); in2.write(L2); in3.write(R1); in4.write(R2);
    }
    void forward()  { raw(1,0,1,0); }
    void backward() { raw(0,1,0,1); }
    void turn_left(){ raw(0,1,1,0); }  // 左轮反转,右轮正转 = 原地左转
    void turn_right(){raw(1,0,0,1); }  // 左轮正转,右轮反转 = 原地右转
    void stop()     { raw(0,0,0,0); }
    void cleanup()  { stop(); in1.close(); in2.close(); in3.close(); in4.close(); }
};
```

## 3.4 pwm.h — 100Hz 软件PWM

```cpp
/*******************************************************************************
* hal/pwm.h — 100Hz 软件PWM (持久fd + clock_gettime 忙等)
* 独立线程运行, 原子变量同步, 支持可配置占空比
*******************************************************************************/
#pragma once
#include <atomic>
#include <thread>
#include "gpio.h"
#include "config.h"

struct pwm_controller {
    gpio_fd pin[4];
    std::atomic<bool>  enabled{false};
    std::atomic<bool>  running{true};
    std::atomic<int>   duty_on{4000};   // ON时长 (us), 默认4000=40%
    std::atomic<int>   duty_off{6000};  // OFF时长 (us)
    std::thread        worker;

    // 电机四个IN脚的逻辑值 (前进)
    static constexpr int FWD[4] = {1,0,1,0};

    void init() {
        int pins[] = {MOTOR_L_IN1, MOTOR_L_IN2, MOTOR_R_IN1, MOTOR_R_IN2};
        for(int i=0;i<4;i++) pin[i].open_rw(pins[i]);
    }
    void start() {
        worker = std::thread([this]() {
            while(running.load()) {
                if(!enabled.load()) { usleep(10000); continue; }
                // ON 期间
                for(int i=0;i<4;i++) pin[i].write(FWD[i]);
                delay_us(duty_on.load());
                // OFF 期间
                for(int i=0;i<4;i++) pin[i].write(0);
                delay_us(duty_off.load());
            }
        });
    }
    void set_duty(int on_ms, int off_ms) { duty_on.store(on_ms); duty_off.store(off_ms); }
    void on()  { enabled.store(true); }
    void off() { enabled.store(false); for(int i=0;i<4;i++) pin[i].write(0); }
    void cleanup() { running.store(false); off(); if(worker.joinable()) worker.join(); }
};
```

## 3.5 servo.h — SG90 舵机控制

```cpp
/*******************************************************************************
* hal/servo.h — SG90 位置舵机 (0-180°, 50Hz PWM)
* 脉冲范围: 600µs(0°) ~ 2400µs(180°)
* 关键: 舵机到位后需稳定等待 400ms 再读数
*******************************************************************************/
#pragma once
#include "gpio.h"
#include "config.h"

struct servo_sg90 {
    gpio_fd sig;

    void init() {
        gpio_quick::setup(SERVO_PIN);
        sig.open_rw(SERVO_PIN);
    }
    // 发送单个 50Hz 脉冲 (20ms 周期)
    void pulse(int width_us) {
        sig.write(1); delay_us(width_us);
        sig.write(0); delay_us(20000 - width_us);
    }
    // 转到目标角度并保持 duration_ms 毫秒
    void move_to(int deg, int duration_ms) {
        if(deg<0)deg=0; if(deg>180)deg=180;
        int pulse_us = 600 + (2400-600) * deg / 180;
        long long end = us_now() + duration_ms * 1000LL;
        while(us_now() < end) pulse(pulse_us);
    }
    // 单次定位脉冲 (维持当前位置)
    void hold(int deg) {
        if(deg<0)deg=0; if(deg>180)deg=180;
        int pulse_us = 600 + (2400-600) * deg / 180;
        pulse(pulse_us); pulse(pulse_us);  // 发两次确保收到
    }
    void cleanup() { sig.close(); }
};
```

## 3.6 ultrasonic.h — US-100 超声波测距

```cpp
/*******************************************************************************
* hal/ultrasonic.h — US-100 超声波传感器 (GPIO模式)
* 测距原理: ECHO回低→TRIG发15µs脉冲→等ECHO升高→
*           脉宽(µs) × 0.01715 = 距离(cm)
* 声速系数: 343m/s ÷ 2(往返) ÷ 10000 = 0.01715 cm/µs
*******************************************************************************/
#pragma once
#include "gpio.h"
#include "config.h"

struct ultrasonic {
    gpio_fd trig, echo;

    void init() {
        gpio_quick::setup(US_TRIG);
        gpio_quick::setup(US_ECHO);
        // echo 设为输入
        char p[64];
        snprintf(p,64,"/sys/class/gpio/gpio%d/direction",US_ECHO);
        int f=::open(p,O_WRONLY); ::write(f,"in",2); ::close(f);
        trig.open_rw(US_TRIG);
        echo.open_rw(US_ECHO);
        trig.write(0);
    }
    // 快速测距 (舵机不动时, ~60ms/次)
    double quick() {
        // 1. 等 ECHO 回低
        long long dl = us_now() + 100000;
        while(echo.read() && us_now() < dl);
        // 2. 发 TRIG 15µs 脉冲
        trig.write(1); delay_us(15); trig.write(0);
        // 3. 等 ECHO 变高
        dl = us_now() + 100000;
        while(!echo.read() && us_now() < dl);
        if(us_now() >= dl) return 999;
        // 4. 测高电平脉宽
        long long t1 = us_now();
        dl = us_now() + 100000;
        while(echo.read() && us_now() < dl);
        if(us_now() >= dl) return 999;
        return (us_now() - t1) * 0.01715;
    }
    // 舵机到位后测距 (稳定读数)
    double stable_read(int settle_ms = 400) {
        usleep(settle_ms * 1000);  // 等传感器稳定
        return quick();
    }
    void cleanup() { trig.close(); echo.close(); }
};
```

## 4.1 dfplayer.h — DFPlayer Mini MP3 模块

```cpp
/*******************************************************************************
* com/dfplayer.h — DFPlayer Mini MP3 UART 控制
* 协议: 10字节命令帧, 9600bps
* 帧格式: 7E FF 06 CMD 00 DH DL CHK_H CHK_L EF
* 校验和 = -(0xFF + 0x06 + CMD + 0x00 + DH + DL)
*******************************************************************************/
#pragma once
#include <fcntl.h>
#include <termios.h>
#include <unistd.h>
#include <cstdio>
#include "config.h"

struct dfplayer {
    int fd = -1;

    // 发送原始命令帧
    void cmd(unsigned char c, unsigned char dh, unsigned char dl) {
        if(fd<0) return;
        int sum = 0xFF + 0x06 + (int)c + 0x00 + (int)dh + (int)dl, chk = -sum;
        unsigned char b[10] = {
            0x7E, 0xFF, 0x06, c, 0x00, dh, dl,
            (unsigned char)((chk>>8)&0xFF),
            (unsigned char)(chk&0xFF), 0xEF
        };
        ::write(fd, b, 10);
        tcdrain(fd);
    }
    // 高级API
    void play(int track) { cmd(0x03, track>>8, track&0xFF); }
    void set_vol(int v)  { cmd(0x09, 0, v&0xFF); }
    void stop()          { cmd(0x0E, 0, 0); }
    void reset()         { cmd(0x0C, 0, 0); usleep(500000); }

    void init() {
        system("killall -9 getty 2>/dev/null");
        usleep(300000);
        fd = ::open(UART_MP3, O_RDWR | O_NOCTTY);
        if(fd<0) { printf("[MP3] 未连接\n"); return; }
        struct termios t; tcgetattr(fd, &t);
        cfsetospeed(&t, B9600); cfsetispeed(&t, B9600);
        t.c_cflag |= (CLOCAL | CREAD);
        t.c_cflag &= ~(PARENB | CSTOPB | CSIZE);
        t.c_cflag |= CS8;
        tcsetattr(fd, TCSANOW, &t);
        tcflush(fd, TCIOFLUSH);
        usleep(1500000);   // 等DFPlayer上电稳定
        reset();           // 硬件复位 (解决模块卡死)
        set_vol(25);
        printf("[MP3] 就绪\n");
    }
    void cleanup() { if(fd>=0) ::close(fd); }
};
```

## 4.2 gsm_4g.h — 4G 短信与通话

```cpp
/*******************************************************************************
* com/gsm_4g.h — 合宙 Air780E 4G Cat.1 AT指令通信
* 功能: 短信发送 + 语音通话
* SMS流程: AT+CMGF=1 → AT+CMGS="号码" → 消息+0x1A → OK
* 通话流程: ATD号码; (分号=语音) → 通话 → ATH 挂断
*******************************************************************************/
#pragma once
#include <stdio.h>
#include <string.h>
#include <fcntl.h>
#include <unistd.h>
#include <termios.h>
#include <time.h>
#include <sys/select.h>
#include "config.h"

struct gsm_4g {
    int fd = -1;

    void init() {
        system("killall -9 getty 2>/dev/null"); usleep(300000);
        fd = ::open(UART_GSM, O_RDWR | O_NOCTTY);
        if(fd<0) { printf("[4G] 模块未连接\n"); return; }
        struct termios t; tcgetattr(fd, &t);
        cfsetospeed(&t, B115200); cfsetispeed(&t, B115200);
        t.c_cflag |= CLOCAL | CREAD;
        t.c_cflag &= ~(PARENB | CSTOPB | CSIZE); t.c_cflag |= CS8;
        tcsetattr(fd, TCSANOW, &t); tcflush(fd, TCIOFLUSH);
        usleep(3000000);  // 等4G模块启动
        send_cmd("AT"); wait_ok(5000);
        send_cmd("AT+CLVL=100"); wait_ok(1000);  // 通话音量最大
        printf("[4G] 就绪\n");
    }
    void send_cmd(const char* c) {
        ::write(fd, c, strlen(c));
        ::write(fd, "\r\n", 2);
        tcdrain(fd);
    }
    int wait_for(const char* expect, int to_ms) {
        char buf[256] = {0}; int t = 0;
        time_t s = time(0);
        while(time(0) - s < to_ms/1000) {
            usleep(100000);
            char tmp[32] = {0}; int n = ::read(fd, tmp, 31);
            if(n>0) { strncat(buf, tmp, sizeof(buf)-t-1); t+=n;
                      if(strstr(buf, expect)) return 1; }
        }
        return 0;
    }
    int wait_ok(int to_ms) { return wait_for("OK", to_ms); }

    // 发送短信
    int send_sms(const char* phone, const char* msg) {
        printf("[4G] 发短信到 %s\n", phone);
        send_cmd("AT+CMGF=1");
        if(!wait_ok(3000)) return -1;
        char c[64]; snprintf(c,64,"AT+CMGS=\"%s\"",phone);
        send_cmd(c);
        if(!wait_for(">", 5000)) return -1;
        char full[512]; snprintf(full,512,"%s\r\n\x1A",msg);
        ::write(fd, full, strlen(msg)+3);
        return wait_ok(10000) ? 0 : -1;
    }
    // 拨打电话
    int dial(const char* phone) {
        char c[64]; snprintf(c,64,"ATD%s;",phone);
        send_cmd(c);
        printf("[4G] 拨号 %s\n", phone);
        return wait_ok(15000) ? 0 : -1;
    }
    void hangup() { send_cmd("ATH"); wait_ok(2000); }
    void cleanup() { if(fd>=0) { hangup(); ::close(fd); } }
};
```

## 4.3 sound_sens.h — 噪音检测

```cpp
/*******************************************************************************
* com/sound_sens.h — KY-038 数字声音传感器
* 防抖策略: 10次采样中 ≥6次超阈值 → 真噪音
* 采样间隔 50ms → 总窗口 500ms
*******************************************************************************/
#pragma once
#include "gpio.h"
#include "config.h"

struct sound_sensor {
    gpio_fd pin;      // DO脚, HIGH=有声

    void init() {
        // 设为输入
        char p[64];
        snprintf(p,64,"/sys/class/gpio/gpio%d/direction",SOUND_PIN);
        int f=::open(p,O_WRONLY); ::write(f,"in",2); ::close(f);
        pin.open_rw(SOUND_PIN);
    }
    bool is_noisy() { return pin.read() == 1; }

    // 10采6中防抖
    bool detect() {
        int hits = 0;
        for(int i=0; i<10; i++) {
            if(is_noisy()) hits++;
            usleep(50000);  // 50ms间隔
        }
        return hits >= 6;
    }
```

## 5.2 fire_detect.h — 火焰检测

```cpp
/*******************************************************************************
* vis/fire_detect.h — 基于HSV颜色空间 + 闪烁特征判别的火焰检测
*
* 算法原理:
*   1. BGR → HSV 色彩空间转换
*   2. 四通道颜色掩码: 红(0-10°含160-179°跨边界) + 橙(10-25°) + 黄(25-35°)
*      S > FIRE_S_MIN(150) 过滤低饱和度物体
*      V > FIRE_V_MIN(150) 过滤暗色物体
*   3. 帧间闪烁判定: |当前帧 - 上一帧| / 上一帧 > FIRE_FLICKER(0.4)
*      真火特征: 像素数大幅波动 (>40%变化率)
*      干扰排除: 人手/灯光/红色物体 变化率 < 20%
*   4. 双条件同时满足 → 确认真实火焰
*
* 实测数据 (LS2K0300 1GHz):
*   单帧检测耗时: < 50ms
*   检测频率: 每 DETECT_INTERVAL(5) 帧 ~3Hz
*   打火机火焰: ~2000px → 检测成功
*   人手/红布:   ~1500px → 闪烁率<10% → 排除
*******************************************************************************/
#pragma once
#include <opencv2/opencv.hpp>
#include <cmath>
#include "config.h"

struct fire_detector {
    int prev_px = 0;
    int curr_px = 0;
    int count   = 0;

    // 返回值: true=检测到火焰
    bool detect(cv::Mat& frame) {
        count++;
        if(count % DETECT_INTERVAL != 0)
            return curr_px > FIRE_PX_MIN;  // 复用上次结果

        cv::Mat hsv, m1, m2, m3, m4, mask;
        cv::cvtColor(frame, hsv, cv::COLOR_BGR2HSV);

        // 四通道火焰颜色 (注意OpenCV中红色跨H边界: 0-10 和 160-179)
        cv::inRange(hsv, cv::Scalar(0,   FIRE_S_MIN, FIRE_V_MIN),
                         cv::Scalar(10,  255, 255), m1);     // 红(前)
        cv::inRange(hsv, cv::Scalar(160, FIRE_S_MIN, FIRE_V_MIN),
                         cv::Scalar(179, 255, 255), m2);     // 红(后,跨边界) [关键!]
        cv::inRange(hsv, cv::Scalar(10,  FIRE_S_MIN, FIRE_V_MIN),
                         cv::Scalar(25,  255, 255), m3);     // 橙
        cv::inRange(hsv, cv::Scalar(25,  FIRE_S_MIN, FIRE_V_MIN),
                         cv::Scalar(35,  255, 255), m4);     // 黄
        mask = m1 | m2 | m3 | m4;

        curr_px = cv::countNonZero(mask);
        int change = abs(curr_px - prev_px);
        float ratio = (prev_px > 0) ? (float)change / prev_px : 0;
        prev_px = curr_px;

        // 双条件: 像素够多 + 闪烁特征
        return curr_px > FIRE_PX_MIN && ratio > FIRE_FLICKER;
    }
    bool is_active() { return curr_px > FIRE_PX_MIN; }
};
```

```cpp
/*******************************************************************************
```

## 5.4 odometry.h — 视觉里程计

```cpp
/*******************************************************************************
* vis/odometry.h — Lucas-Kanade 光流法视觉里程计
*
* 算法原理:
*   1. goodFeaturesToTrack 提取 100 个 FAST 角点
*   2. calcOpticalFlowPyrLK 金字塔光流跟踪到下一帧
*   3. 统计水平位移 avg_dx → 航向角变化 (标定系数 0.1°/像素)
*   4. 过滤: |dx|>50 异常值丢弃, cnt<5 无效帧忽略
*
* 定位精度: 3m×4m室内累积航向 ±5°, 长距离累计误差 30-50cm
* 替代方案: MPU6050 陀螺仪 (I2C, 5元) 可实现更高精度
*******************************************************************************/
#pragma once
#include <opencv2/opencv.hpp>
#include <cmath>
#include <vector>

struct visual_odometry {
    std::vector<cv::Point2f> prev_pts;
    cv::Mat                  prev_gray;
    double heading = 0, pos_x = 0, pos_y = 0;

    void update(cv::Mat& gray) {
        // 1. 提取 FAST 角点
        std::vector<cv::Point2f> new_pts;
        cv::goodFeaturesToTrack(gray, new_pts, 100, 0.01, 10);

        if(!prev_pts.empty() && !new_pts.empty()) {
            // 2. LK 光流跟踪
            std::vector<cv::Point2f> tracked;
            std::vector<unsigned char> status;
            std::vector<float> err;
            cv::calcOpticalFlowPyrLK(prev_gray, gray, prev_pts,
                tracked, status, err);

            // 3. 统计水平位移
            double sum_dx = 0; int cnt = 0;
            for(size_t i=0; i<status.size(); i++) {
                if(!status[i]) continue;
                double dx = tracked[i].x - prev_pts[i].x;
                if(fabs(dx) > 50) continue;  // 过滤误匹配
                sum_dx += dx; cnt++;
            }
            // 4. 推算航向角 (0.1°/像素 标定系数)
            if(cnt > 5) {
                double turn = sum_dx / cnt * 0.1;
                heading += turn;
                if(heading < 0) heading += 360;
                if(heading >= 360) heading -= 360;
                double rad = heading * M_PI / 180.0;
                pos_x += 10 * cos(rad);
                pos_y += 10 * sin(rad);
            }
        }
        prev_gray = gray.clone();
        prev_pts  = new_pts;
    }
    void report() {
        printf("[ODO] pos=(%.0f,%.0f) hdg=%.0f°\n", pos_x, pos_y, heading);
    }
};
```

```cpp
#include <arpa/inet.h>
#include <unistd.h>
#include <thread>
#include <mutex>
#include <atomic>
#include <vector>
#include <sstream>
#include "config.h"

struct mjpeg_server {

    std::mutex&         frame_mtx;
```

## 6.3 state_machine.h — 主状态机

```cpp
/*******************************************************************************
* dec/state_machine.h — 巡检小车主状态机
*
* 优先级: 火焰检测 > 人腿检测 > 正常巡检
*
*   发现火焰 → 停车 + 播0003报警 + fire_active=true
*     fire_active && 无火3秒 → fire_active=false
*
*   !fire_active:
*     S_NORMAL → 前进 + 每13s播0001 + Haar检测人腿
*       检测到人腿 → 停车 + 播0002 → S_STOPPED
*     S_STOPPED → 停车 + 3秒无人 → 恢复前进 + 播0001 → S_NORMAL
*******************************************************************************/
#pragma once
#include <ctime>
#include "../com/dfplayer.h"
#include "../vis/fire_detect.h"
#include "../vis/leg_detect.h"
#include "../hal/motor.h"
#include "../hal/pwm.h"
#include "config.h"

enum patrol_state { S_NORMAL, S_STOPPED };

struct patrol_machine {
    dfplayer&        mp3;
    fire_detector&   fire;
    leg_detector&    leg;
    motor&           mt;
    pwm_controller&  pw;

    patrol_state state     = S_NORMAL;
    bool         fire_active = false;
    time_t       fire_last   = 0;
    time_t       last_0001   = 0;
    time_t       stop_time   = 0;

    patrol_machine(dfplayer& m, fire_detector& f, leg_detector& l,
                   motor& mt_, pwm_controller& p)
        : mp3(m), fire(f), leg(l), mt(mt_), pw(p) {}

    void init() {
        mt.forward(); pw.on();
        mp3.play(1);
        last_0001 = time(0);
        printf("[STATE] 开始巡检\n");
    }

    // 每帧调用, 返回 true 表示正常继续
    bool tick(cv::Mat& frame) {
        bool f = fire.detect(frame);
        int  d = leg.detect(frame);
        time_t now = time(0);

        // === 优先级1: 火焰检测 ===
        if(f && !fire_active) {
            printf("🔥 火焰! 停车\n");
            mt.stop(); pw.off();
            mp3.play(3);  // 0003 火焰报警
            fire_active = true;
            fire_last = now;
        }
        if(fire_active && !fire.is_active() && now - fire_last >= 3) {
            printf("🔥 火焰解除\n");
            fire_active = false;
        }

        // 火焰激活期间跳过其他逻辑
        if(fire_active) return true;

        // === 优先级2: 人腿检测 + 正常巡检 ===
        if(state == S_NORMAL) {
            // 每13秒播背景音
            if(now - last_0001 >= 13) {
                mp3.play(1);
                last_0001 = now;
            }
            // 确保电机开着
            pw.on();

            if(leg.has_human()) {
                printf("⚠️ 检测到%d人 → 停车\n", d);
                mt.stop(); pw.off();
                mp3.play(2);  // 0002 请尽快离开
                state = S_STOPPED;
                stop_time = now;
            }
        } else {  // S_STOPPED
            pw.off();
            if(!leg.has_human() && now - stop_time >= 3) {
                printf("✅ 无人,恢复前进\n");
                mt.forward(); pw.on();
                mp3.play(1);
                last_0001 = now;
                state = S_NORMAL;
            }
        }

        // 每秒输出状态
        static time_t lp = 0;
        if(now - lp >= 1) {
            printf("[%s] 人:%d\n", state==S_NORMAL?"正常":"停车", d);
            lp = now;
        }
        return true;
    }
};
```

# 七、主程序

## main.cpp

```cpp
/*******************************************************************************
* main.cpp — LS2K0300 全自动巡检小车 统一主程序
*
* 模块化架构:
*   HAL (硬件抽象层) — gpio / motor / pwm / servo / ultrasonic
*   COM (通信层)     — dfplayer / gsm_4g / sound_sensor / voice_wake
*   VIS (视觉层)     — camera / fire_detector / leg_detector / odometry
*   DEC (决策层)     — obstacle_avoider / wall_follower / patrol_machine
*
* 多线程:
*   cam_thread    — 摄像头 320×240@15fps
*   speed_thread  — 100Hz 软件PWM
*   mjpeg_thread  — HTTP MJPEG 推流 (端口8081, 调试用)
*   main loop     — 状态机 + 避障  ~12.5Hz
*
* 编译:
*   loongarch64-linux-gnu-g++ -O2 -std=c++17 main.cpp \
*     -I/opt/ls_2k0300_env/opencv_4_10_build/include/opencv4 \
*     -L/opt/ls_2k0300_env/opencv_4_10_build/lib \
*     -lopencv_core -lopencv_imgproc -lopencv_videoio \
*     -lopencv_objdetect -lpthread \
*     -Wl,-rpath-link,/opt/ls_2k0300_env/opencv_4_10_build/lib
*******************************************************************************/

// ==================== HAL ====================
#include "hal/gpio.h"
#include "hal/motor.h"
#include "hal/pwm.h"
#include "hal/servo.h"
#include "hal/ultrasonic.h"

// ==================== COM ====================
#include "com/dfplayer.h"
#include "com/gsm_4g.h"
#include "com/sound_sens.h"
#include "com/voice_wake.h"

// ==================== VIS ====================
#include "vis/camera.h"
#include "vis/fire_detect.h"
#include "vis/leg_detect.h"
#include "vis/odometry.h"
#include "vis/mjpeg.h"

// ==================== DEC ====================
#include "dec/avoid.h"
#include "dec/wall_follow.h"
#include "dec/state_machine.h"

#include <signal.h>
#include <thread>
#include <chrono>

// ==================== 全局对象 ====================
motor          g_motor;
pwm_controller g_pwm;
servo_sg90     g_servo;
ultrasonic     g_ultrasonic;

dfplayer       g_mp3;
gsm_4g         g_gsm;
sound_sensor   g_sound_sens;
voice_wake     g_voice;

camera         g_camera;
fire_detector  g_fire;
leg_detector   g_leg;
visual_odometry g_odo;

patrol_machine g_patrol(g_mp3, g_fire, g_leg, g_motor, g_pwm);

// ==================== 信号处理 (安全退出) ====================
void cleanup_all() {
    printf("\n[EXIT] 安全退出...\n");
    g_pwm.off();
    g_motor.stop();
    g_camera.stop();
    g_motor.cleanup();
    g_pwm.cleanup();
    g_servo.cleanup();
    g_ultrasonic.cleanup();
    g_mp3.cleanup();
    g_gsm.cleanup();
    g_sound_sens.cleanup();
}

void signal_handler(int) {
    cleanup_all();
    exit(0);
}

// ==================== 主函数 ====================
int main() {
    signal(SIGINT, signal_handler);

    printf("========================================\n");
    printf(" LS2K0300 全自动巡检小车 v2.0\n");
    printf(" 模块化架构: HAL→COM→VIS→DEC\n");
    printf("========================================\n\n");

    // 1. 初始化 HAL
    printf("--- HAL 初始化 ---\n");
    g_motor.init();
    g_pwm.init();
    g_pwm.start();
    g_servo.init();
    g_ultrasonic.init();
    printf("HAL: OK\n\n");

    // 2. 初始化 COM
    printf("--- COM 初始化 ---\n");
    g_mp3.init();
    // g_gsm.init();    // 如需4G功能取消注释
    // g_sound_sens.init();
    printf("COM: OK\n\n");

    // 3. 初始化 VIS
    printf("--- VIS 初始化 ---\n");
    g_leg.load();
    g_camera.start();
    sleep(2);  // 等摄像头就绪
    printf("VIS: OK\n\n");

    // 4. 启动 MJPEG 调试推流
    mjpeg_server mjpeg(g_camera.mtx, g_camera.frame_color, &g_leg.cascade);
    mjpeg.start();

    // 5. 初始化状态机
    printf("--- 开始巡检 ---\n");
    g_patrol.init();

    // 6. 主循环
    while(g_camera.running.load()) {
        cv::Mat frame = g_camera.get_color();
        cv::Mat gray  = g_camera.get_gray();
        if(frame.empty()) { usleep(50000); continue; }

        // 视觉检测 + 状态机
        g_patrol.tick(frame);

        // 视觉里程计
        if(!gray.empty()) g_odo.update(gray);

        // 超声波避障 (仅在正常状态且无火情时执行)
        static time_t last_odo = 0;
        time_t now = time(0);
        if(now - last_odo >= 2) {
            g_odo.report();
            last_odo = now;
        }

        usleep(80000);  // ~12.5Hz
    }

    cleanup_all();
    return 0;
}
```
